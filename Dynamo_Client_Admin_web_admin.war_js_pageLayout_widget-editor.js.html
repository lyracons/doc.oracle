/**
 * @fileoverview Defines the controls for a widget editor that can
 * edit the configuration options for a widget instance.
 *
 */
/*global $ */
/*global document */
define(
  //-----------------------------------------------------------------
  // MODULE NAME
  //-----------------------------------------------------------------
  'editPageLayout/widget-editor',
  
  //-----------------------------------------------------------------
  // DEPENDENCIES
  //-----------------------------------------------------------------
  ['knockout'],
    
  //-----------------------------------------------------------------
  // MODULE DEFINITION
  //-----------------------------------------------------------------
  function (ko) {
    "use strict";
    
    //----------------------------------------------------------------
    // Class definition & member variables (the constructor)
    //----------------------------------------------------------------
    /**
       Creates a WidgetEditorViewModel.
       @private
       @name WidgetEditorViewModel
       @property {observable<WidgetViewModel>} widget The widget instance where changes should be saved to.
       @property {observable<WidgetViewModel>} editWidget A copy of widget to hold changes before they are saved.
       @property {string} template The name of the widget editor template.
       @property {string} path The url path to the widget editor template.
       @property {string} saveString The string label for the save button.
       @property {string} closeString The string label for the close button.
       @property {observable<boolean>} minimized Whether or not the editor is minimized
       @class The WidgetEditorViewModel is responsible for controlling and providing the ability to edit a widget instance's
       configuration options.
     */
    function WidgetEditorViewModel() {
      //The widget and the edit widget for editing and rollback
      //capabilities
      this.widget = ko.observable();
      this.editWidget = ko.observable();
      
      //Path and template for the editor
      this.template = 'widget-edit-template';
      this.path = '';
      
      //Display resources
      this.saveString = 'Save';
      this.closeString = 'Cancel';
      
      //Minimization controls
      this.minimized = ko.observable(true);
      
      return(this);
    }
    
    //----------------------------------------------------------------
    // Class functions
    //----------------------------------------------------------------
    
    /**
       Expands the editor out from its minimized view. Will also initialize the editWidget to hold
       unsaved changes.
       @private
     */
    WidgetEditorViewModel.prototype.expandEditor = function() {
      
      if(this.widget()) {
        //Update the temporary editing widget
        this.editWidget(this.initializeEdit(this.widget()));
      }
      
      //Show the editor
      this.minimized(false);
      $('#editor-CC').modal('show');
      $(document).off('focusin.modal');
    };
        
    
    /**
       Minimizes the editor reverting any unsaved changed in the process.
       @private
     */
    WidgetEditorViewModel.prototype.minimizeEditor = function () {
      this.editWidget(null); //Don't need to track the editWidget anymore.
      
      //Close the editor
      this.minimized(true);
      $('#editor-CC').modal('hide');
    };
    
    /**
       Cancels any unsaved changes without closing the editor.
       @private
     */
    WidgetEditorViewModel.prototype.cancelChanges = function() {
      if(this.widget()) {
        //Reset the edit widget to the widget's properties
        this.editWidget(this.initializeEdit(this.widget()));
      }
    };
    
    /**
       Saves any changes to the editWidget persisting them back to the widget
       instance. 
       @private
     */
    WidgetEditorViewModel.prototype.saveChanges = function () {
      if(this.widget()) {
        //Update the widget
        this.acceptEdit(this.widget(), this.editWidget());
      }
    };
    
    /**
       Sets a widget as the widget to edit with the editor. Also opens the editor.
       @private
       @param {WidgetViewModel} widget The widget instance to edit.
     */
    WidgetEditorViewModel.prototype.setWidget = function(widget) {
      this.widget(widget);
      this.expandEditor();
    };
    
    /**
       Initializes the editWidget property to be a copy of the widget where unsaved changes can be stored.
       @private
       @param {WidgetViewModel} widget The widget instance to copy for editing.
       @return {WidgetViewModel} A copy of the widget that can be used for editing.
     */
    WidgetEditorViewModel.prototype.initializeEdit = function(widget) {
      var editWidget = {};
      var key;
          
      //Copy observables from the widget to an editWidget that will hold temporary changes
      for(key in widget) {
        if(widget.hasOwnProperty(key) && ko.isObservable(widget[key]) && !ko.isComputed(widget[key])) {
          
          //Only observableArrays have a remove function
          if(widget[key].remove) {
            editWidget[key] = ko.observableArray(widget[key]());
          } else {
            editWidget[key] = ko.observable(widget[key]());
          }
        }
      }
      return editWidget;
    };
    
    /**
       Accepts changes to the editWidget and persists them back to the widget.
       @private
       @param {WidgetViewModel} widget The widget instance to persists the changes to.
       @param {WidgetViewModel} editWidget The editWidget that contains the changes.
     */
    WidgetEditorViewModel.prototype.acceptEdit = function(widget, editWidget) {
      var key;
      for(key in widget) {
        
        //Copy any observable non-computed properties back to the widget
        //Undefined properties are assumed to be properties populated onLoad or by some other means rather
        //than the editor
        if(widget.hasOwnProperty(key) && ko.isObservable(widget[key]) && !ko.isComputed(widget[key]) &&
           widget[key]() !== editWidget[key]() && typeof editWidget[key]() !== 'undefined' && !widget[key].isData) {
          widget[key](editWidget[key]());
        }
      }
    };
    
    return WidgetEditorViewModel;
  }
);
